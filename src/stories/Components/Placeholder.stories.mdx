# Placeholder

Used to show in place of another element that is loading.

[<Button size='medium'>View Source</Button>](https://github.com/smurph7/stitch-ui/blob/main/src/components/Placeholder.tsx)

export const Template = args => <Placeholder {...args} />;

### Usage

<Canvas>
  <Story name="Small" args={{ width: 200, height: 20 }}>
    {Template.bind({})}
  </Story>
  <Story name="Large" args={{ width: 150, height: 60 }}>
    {Template.bind({})}
  </Story>
  <Story
    name="Round"
    args={{ width: 100, height: 100, borderRadius: '$round' }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

<Canvas>
  <Story
    name="Background"
    args={{
      width: 100,
      height: 100,
      bg: '$green5',
      background:
        'linear-gradient(to left, $green2, $green3, $green4, $green5, $green6)'
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
