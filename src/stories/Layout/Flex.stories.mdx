# Flex

Flex is `Box` with `display: flex` and comes with helpful style shorthand. It renders a `div` element.

[<Button size='medium'>View Source</Button>](https://github.com/smurph7/stitch-ui/blob/main/src/components/Flex.tsx)

export const Template = args => <Flex {...args} />;

export const RowTemplate = args => (
  <Flex css={{ height: '150px' }} {...args}>
    <Flex
      align="center"
      justify="center"
      css={{ bg: '$green5', width: '30%', height: '100%' }}
    >
      Box 1
    </Flex>
    <Flex
      align="center"
      justify="center"
      css={{ bg: '$blue5', width: '20%', height: '100%' }}
    >
      Box 2
    </Flex>
    <Flex css={{ bg: '$tomato5', width: '10%', height: '100%' }}>Box 3</Flex>
  </Flex>
);

export const ColumnTemplate = args => (
  <Flex direction="column" css={{ height: '150px' }} {...args}>
